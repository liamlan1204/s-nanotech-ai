  <script>
    // Inicijalizacija Pi SDK
    const scopes = ['username', 'payments'];
    const onIncompletePaymentFound = (payment) => {
      console.log('Incomplete payment found', payment);
    };

    Pi.init({ version: "2.0", sandbox: true }); // za Testnet

    function loginWithPi() {
      Pi.authenticate(scopes, onIncompletePaymentFound).then(function(auth) {
        document.getElementById("response").innerHTML =
          "Welcome, " + auth.user.username + "!<br>Processing your demo payment...";

        // Simulacija Pi plaćanja – u pravoj verziji bi ovdje bila funkcija Pi.createPayment()
        setTimeout(() => {
          document.getElementById("response").innerHTML += "<br>✔ Access granted. AI assistant ready.";
        }, 2000);
      }).catch(function(error) {
        console.error(error);
        alert("Authentication failed.");
      });
    }
